228 /*****************************************************************/
229 /* UDP framing format */
230 /* ----------------------- */
231 /* Several people have asked about dissecting MAC by framing */
232 /* PDUs over IP. A suggested format over UDP has been created */
233 /* and implemented by this dissector, using the definitions */
234 /* below. A link to an example program showing you how to encode */
235 /* these headers and send LTE MAC PDUs on a UDP socket is */
236 /* provided at https://wiki.wireshark.org/MAC-LTE */
237 /* */
238 /* A heuristic dissector (enabled by a preference) will */
239 /* recognise a signature at the beginning of these frames. */
240 /*****************************************************************/
241 
242 
243 /* Signature. Rather than try to define a port for this, or make the
244  port number a preference, frames will start with this string (with no
245  terminating NULL */
246 #define MAC_LTE_START_STRING "mac-lte"
247 
248 /* Fixed fields. This is followed by the following 3 mandatory fields:
249  - radioType (1 byte)
250  - direction (1 byte)
251  - rntiType (1 byte)
252  (where the allowed values are defined above */
253 
254 /* Optional fields. Attaching this info to frames will allow you
255  to show you display/filter/plot/add-custom-columns on these fields, so should
256  be added if available.
257  The format is to have the tag, followed by the value (there is no length field,
258  it's implicit from the tag) */
259 
260 #define MAC_LTE_RNTI_TAG 0x02
261 /* 2 bytes, network order */
262 
263 #define MAC_LTE_UEID_TAG 0x03
264 /* 2 bytes, network order */
265 
266 #define MAC_LTE_FRAME_SUBFRAME_TAG 0x04
267 /* 2 bytes, network order, SFN is stored in 12 MSB and SF in 4 LSB */
268 
269 #define MAC_LTE_PREDEFINED_DATA_TAG 0x05
270 /* 1 byte */
271 
272 #define MAC_LTE_RETX_TAG 0x06
273 /* 1 byte */
274 
275 #define MAC_LTE_CRC_STATUS_TAG 0x07
276 /* 1 byte */
277 
278 #define MAC_LTE_EXT_BSR_SIZES_TAG 0x08
279 /* 0 byte */
280 
281 #define MAC_LTE_SEND_PREAMBLE_TAG 0x09
282 /* 2 bytes, RAPID value (1 byte) followed by RACH attempt number (1 byte) */
283 
284 #define MAC_LTE_CARRIER_ID_TAG 0x0A
285 /* 1 byte */
286 
287 #define MAC_LTE_PHY_TAG 0x0B
288 /* variable length, length (1 byte) then depending on direction
289  in UL: modulation type (1 byte), TBS index (1 byte), RB length (1 byte),
290  RB start (1 byte), HARQ id (1 byte), NDI (1 byte)
291  in DL: DCI format (1 byte), resource allocation type (1 byte), aggregation level (1 byte),
292  MCS index (1 byte), redundancy version (1 byte), resource block length (1 byte),
293  HARQ id (1 byte), NDI (1 byte), TB (1 byte), DL reTx (1 byte) */
294 
295 #define MAC_LTE_SIMULT_PUCCH_PUSCH_PCELL_TAG 0x0C
296 /* 0 byte */
297 
298 #define MAC_LTE_SIMULT_PUCCH_PUSCH_PSCELL_TAG 0x0D
299 /* 0 byte */
300 
301 #define MAC_LTE_CE_MODE_TAG 0x0E
302 /* 1 byte containing mac_lte_ce_mode enum value */
303 
304 #define MAC_LTE_NB_MODE_TAG 0x0F
305 /* 1 byte containing mac_lte_nb_mode enum value */
306 
307 #define MAC_LTE_N_UL_RB_TAG 0x10
308 /* 1 byte containing the number of UL resource blocks: 6, 15, 25, 50, 75 or 100 */
309 
310 #define MAC_LTE_SR_TAG 0x11
311 /* 2 bytes for the number of items, followed by that number of ueid, rnti (2 bytes each) */
312 
313 
314 /* MAC PDU. Following this tag comes the actual MAC PDU (there is no length, the PDU
315  continues until the end of the frame) */
316 #define MAC_LTE_PAYLOAD_TAG 0x01
