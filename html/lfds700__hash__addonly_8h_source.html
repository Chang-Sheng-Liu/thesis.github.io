9  lfds700_pal_uint_t \
10  loop; \
11  \
12  for( loop = 0 ; loop < (data_length_in_bytes) ; loop++ ) \
13  { \
14  (hash) += *( (char unsigned *) (data) + loop ); \
15  (hash) += ((hash) << 10); \
16  (hash) ^= ((hash) >> 6); \
17  } \
18  \
19  (hash) += ((hash) << 3); \
20  (hash) ^= ((hash) >> 11); \
21  (hash) += ((hash) << 15); \
22  }
